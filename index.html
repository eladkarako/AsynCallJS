Example First

function sqr(n) {
  return Math.pow(n, 2);
}

sqr.asyncall_callback = function (result, args) {
  console.log("parameters used:", args);
  console.log("result:", result);
};

sqr.asyncall(5);

//output #1:  "parameters used:" [5]
//output #2:  "result:" 25
